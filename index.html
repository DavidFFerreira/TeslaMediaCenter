tesla-bypass.js — Código completo 
// ─────────────────────────────────────────────
// Tesla Drive Mode Video Bypass
// Funciona no browser Chromium embebido do Tesla
// ─────────────────────────────────────────────

(function TeslaBypass() {
  const video = document.querySelector('video');
  let audioCtx = null;
  let wakeLock = null;
  let userPaused = false;

  // 1. Iniciar AudioContext após interação
  function initAudio() {
    if (audioCtx) return;
    audioCtx = new (window.AudioContext || window.webkitAudioContext)();
    const src = audioCtx.createMediaElementSource(video);
    src.connect(audioCtx.destination);
    updateStatus('audio-state', 'ATIVO');
  }

  document.addEventListener('click', initAudio, { once: true });

  // 2. Wake Lock — mantém ecrã ativo
  async function requestWakeLock() {
    try {
      wakeLock = await navigator.wakeLock.request('screen');
      updateStatus('wake-state', 'ATIVO');
    } catch (e) {
      updateStatus('wake-state', 'N/D');
    }
  }
  requestWakeLock();

  // 3. O truque principal — visibilitychange
  // O Tesla dispara 'hidden' quando o carro entra em marcha
  document.addEventListener('visibilitychange', () => {
    const state = document.visibilityState;
    updateStatus('vis-state', state.toUpperCase());

    if (state === 'hidden') {
      // Retoma áudio e vídeo
      if (audioCtx?.state === 'suspended') audioCtx.resume();
      video.play().catch(() => {});
    } else {
      // Recupera wake lock quando visível de novo
      if (!wakeLock || wakeLock.released) requestWakeLock();
    }
  });

  // 4. Interceta pausa não intencional
  video.addEventListener('pause', () => {
    if (!userPaused) {
      setTimeout(() => video.play().catch(() => {}), 80);
    }
  });

  // Detectar pausa intencional do utilizador
  video.addEventListener('click', () => {
    userPaused = video.paused ? false : true;
  });

})();
