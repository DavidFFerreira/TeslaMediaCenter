(function TeslaBypass() {
  'use strict';

  // ── CONFIG
  const RETRY_INTERVAL = 200; // ms entre tentativas de play
  const RETRY_DURATION = 3000; // ms máximo de retry após visibilitychange

  let audioCtx = null;
  let wakeLock = null;
  let userPaused = false;
  let retryTimer = null;
  let ytPlayer = null; // referência ao YouTube IFrame Player

  // ── UTILS
  function tryPlay(video) {
    if (!video || userPaused) return;
    video.play().catch(() => {});
  }

  function tryPlayYT() {
    if (ytPlayer && typeof ytPlayer.playVideo === 'function') {
      ytPlayer.playVideo();
    }
  }

  // ── 1. AudioContext — pipeline de áudio ativo em background
  //    CORREÇÃO: ligar ao vídeo específico na primeira interação
  //    com esse elemento, não num click genérico.
  function initAudio(video) {
    if (audioCtx) return;
    try {
      audioCtx = new (window.AudioContext || window.webkitAudioContext)();
      const src = audioCtx.createMediaElementSource(video);
      src.connect(audioCtx.destination);
      updateUI();
    } catch(e) { /* browser pode não suportar */ }
  }

  // ── 2. Wake Lock — impede ecrã de adormecer
  async function requestWakeLock() {
    try {
      if (navigator.wakeLock) {
        wakeLock = await navigator.wakeLock.request('screen');
        wakeLock.addEventListener('release', () => {
          // Recupera quando a tab volta a ficar visível
          if (document.visibilityState === 'visible') requestWakeLock();
        });
        updateUI();
      }
    } catch(e) { /* API não suportada no browser do Tesla mais antigo */ }
  }
  requestWakeLock();

  // ── 3. O BYPASS PRINCIPAL ────────────────────────────────────

// BUG 1
YouTube iframe

O iframe do YouTube é cross-origin. video.play() num iframe não funciona. Corrigido com a YouTube IFrame API.

// BUG 2
userPaused invertido

A lógica original estava ao contrário — pausava quando devia tocar e vice-versa. Corrigido com registo de intenção antes do evento.

AudioContext genérico

O AudioContext iniciava no primeiro click na página, não no vídeo. Corrigido para iniciar no evento play do elemento correto.
