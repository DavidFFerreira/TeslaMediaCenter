// Tesla Drive Mode Video Bypass — tesla-bypass.js
// Funciona no browser Chromium embebido do Tesla
// ─────────────────────────────────────────────────────

(function TeslaBypass() {
  const video = document.querySelector('video');
  let audioCtx = null;
  let wakeLock = null;
  let userPaused = false;

  // ── 1. AudioContext — mantém pipeline de áudio ativo
  function initAudio() {
    if (audioCtx) return;
    audioCtx = new (window.AudioContext || window.webkitAudioContext)();
    const src = audioCtx.createMediaElementSource(video);
    src.connect(audioCtx.destination);
  }
  // Requer interação do utilizador para iniciar
  document.addEventListener('click', initAudio, { once: true });

  // ── 2. Wake Lock — impede o ecrã de adormecer
  async function requestWakeLock() {
    try {
      wakeLock = await navigator.wakeLock.request('screen');
      wakeLock.addEventListener('release', () => {
        // Recupera o wake lock quando possível
        if (document.visibilityState === 'visible') requestWakeLock();
      });
    } catch (e) { /* API não suportada */ }
  }
  requestWakeLock();

  // ── 3. O TRUQUE PRINCIPAL ─────────────────────────
  // O Tesla dispara visibilitychange → 'hidden'
  // quando o carro sai do lugar (entra em Drive/Reverse).
  // O browser pausa automaticamente qualquer <video>.
  // Intercetamos e forçamos a retoma imediata.
  document.addEventListener('visibilitychange', () => {
    if (document.visibilityState === 'hidden') {
      // Retoma o AudioContext se suspenso
      if (audioCtx?.state === 'suspended') audioCtx.resume();
      // Força retoma do vídeo
      video.play().catch(() => {});
    } else {
      // Recupera wake lock quando visível
      if (!wakeLock || wakeLock.released) requestWakeLock();
    }
  });

  // ── 4. Interceta pausas involuntárias
  // Além do visibilitychange, o Tesla pode tentar pausar
  // o vídeo por outros meios. Este handler garante a retoma.
  video.addEventListener('pause', () => {
    if (!userPaused) {
      setTimeout(() => video.play().catch(() => {}), 80);
    }
  });

  // Distingue pausa do utilizador vs pausa do sistema
  video.addEventListener('click', () => {
    userPaused = video.paused ? false : true;
  });

  // ── 5. PLEX — fetch com token direto ao servidor
  async function plexFetch(serverUrl, token, path) {
    const url = `${serverUrl}${path}?X-Plex-Token=${token}`;
    const res = await fetch(url, {
      headers: {
        'Accept': 'application/json',
        'X-Plex-Token': token,
        'X-Plex-Product': 'Tesla Media Center',
        'X-Plex-Device': 'Tesla',
      }
    });
    return res.json();
  }

  // Obter URL de stream direto de um item Plex
  async function getPlexStream(serverUrl, token, ratingKey) {
    const data = await plexFetch(serverUrl, token,
      `/library/metadata/${ratingKey}`);
    const part = data?.MediaContainer?.Metadata?.[0]?.Media?.[0]?.Part?.[0];
    if (!part?.key) throw new Error('Stream não encontrado');
    return `${serverUrl}${part.key}?X-Plex-Token=${token}`;
  }

})();
